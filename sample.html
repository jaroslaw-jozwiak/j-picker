JPikcer with default config

const jpicker = new JPicker({
    wrapper: '.date-picker-default-wrapper'
});

JPikcer with own config

const jpicker = new JPicker({
    wrapper: '.date-picker-config-wrapper',
    title: 'Your title',
    description: 'Some text',
    range: true,
    currentDate: [new Date(2022, 4, 10), new Date(2022, 4, 15)],
    today: new Date(2022, 4, 15, 1, 22),
    weekDays: [
        'M', 'T', 'W', 'T', 'F', 'Sa', 'Su'
    ],
    months: [
        'I', 'II', 'III', 'IV', 'V', 'VI', 'VII', 'VIII', 'IX', 'X', 'XI', 'XII'
    ],
    events: {
        onValueClick: function() {
            console.log('onValueClick', arguments)
        },
        onDayClick: function() {
            console.log('onDayClick', arguments)
        },
        onChangeValue: function() {
            console.log('onChangeValue', arguments)
        }
    }
});
